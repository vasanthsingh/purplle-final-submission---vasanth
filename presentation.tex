\section{Problem Statement}
% ══════════════════════════════════════════

\begin{frame}{Problem Statement}
  \begin{block}{Objective}
    Transform raw CCTV footage from a Purplle retail store into
    \textbf{structured behavioural events} and \textbf{real-time analytics}.
  \end{block}

  \vspace{0.4cm}

  \begin{columns}[T]
    \begin{column}{0.45\textwidth}
      \textbf{Input}
      \begin{itemize}
        \item 5 CCTV clips (1080p, $\sim$20 min each)
        \item Brigade Road, Bangalore store
        \item Real POS transaction data
        \item Store floor plan
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Output}
      \begin{itemize}
        \item Footfall and unique visitor count
        \item Conversion funnel (Entry to Purchase)
        \item Zone dwell-time heatmaps
        \item Queue depth and abandonment rates
        \item Real-time anomaly alerts
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{CV Pipeline}
% ══════════════════════════════════════════

\begin{frame}{CV Pipeline -- Detection and Tracking}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{Person Detection}
      \begin{itemize}
        \item Model: YOLOv8-m (Medium)
        \item Resolution: \texttt{imgsz=320} for CPU speed
        \item Confidence: \texttt{conf=0.20}
        \item Sampling: 5 fps subsampling
      \end{itemize}

      \vspace{0.3cm}
      \textbf{Multi-Object Tracking}
      \begin{itemize}
        \item ByteTrack via \texttt{supervision} library
        \item Persistent track IDs per camera
        \item Continuous visitor trajectories
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{block}{Results (Brigade Dataset)}
        \centering
        \begin{tabular}{lr}
          \toprule
          \textbf{Metric} & \textbf{Value} \\
          \midrule
          Frames processed & 19,515 \\
          Events emitted & 193 \\
          Unique visitors & 56 \\
          Staff flagged & 55 \\
          REENTRY events & 9 \\
          Avg confidence & 0.710 \\
          Total time & $\sim$46 min \\
          \bottomrule
        \end{tabular}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{CV Pipeline -- Event Generation}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Zone Events}
      \begin{itemize}
        \item Entry/Exit: Signed line crossing
        \item Zone Enter/Exit: Point-in-polygon test
        \item Zone Dwell: 30-second cadence timer
        \item Billing Queue: Join/Leave/Abandon
      \end{itemize}

      \vspace{0.3cm}
      \textbf{Staff Detection}
      \begin{itemize}
        \item Primary: HSV uniform colour match
        \item Secondary: Dwell-pattern heuristic
        \item Stockroom: forced staff flag
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Re-entry Detection}
      \begin{itemize}
        \item 5-minute sliding appearance buffer
        \item 3-bin HSV histogram signature
        \item Cosine similarity threshold: 0.90
        \item Self-pruning cache
      \end{itemize}

      \vspace{0.3cm}
      \textbf{Cross-Camera Dedup}
      \begin{itemize}
        \item Key: (visitor\_id, zone\_id)
        \item 3-second suppression window
        \item Auto-prunes every 50 frames
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{API Design}
% ══════════════════════════════════════════

\begin{frame}{API Endpoints}
  \centering
  \small
  \begin{tabular}{lll}
    \toprule
    \textbf{Method} & \textbf{Path} & \textbf{Description} \\
    \midrule
    GET & /health & Service health + staleness \\
    POST & /events/ingest & Batch event ingest (idempotent) \\
    POST & /pos/ingest & POS transaction ingest \\
    GET & /stores/\{id\}/metrics & Footfall, conversion, queue depth \\
    GET & /stores/\{id\}/funnel & Conversion pipeline stages \\
    GET & /stores/\{id\}/heatmap & Zone intensity (0--100) \\
    GET & /stores/\{id\}/anomalies & Alerts with severity \\
    GET & /stores/\{id\}/events & Paginated raw event list \\
    WS & /ws/stores/\{id\} & Real-time WebSocket stream \\
    \bottomrule
  \end{tabular}
\end{frame}

\begin{frame}{Idempotent Partial-Success Ingest}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Response Codes}
      \begin{itemize}
        \item \textbf{200} -- All events accepted
        \item \textbf{207} -- Some rejected (partial success)
        \item \textbf{413} -- Batch exceeds 500 limit
      \end{itemize}

      \vspace{0.3cm}
      \textbf{Idempotency}
      \begin{itemize}
        \item UUID primary key on event\_id
        \item ON CONFLICT DO NOTHING
        \item Pipeline retries are always safe
      \end{itemize}

      \vspace{0.3cm}
      \textbf{Two-Phase Abandon}
      \begin{enumerate}
        \item Pipeline emits best-guess LEAVE/ABANDON
        \item API reclassifies after POS ingest
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Response Example}
\begin{lstlisting}[language={}]
{
  "accepted": 497,
  "duplicates": 2,
  "rejected": [
    {
      "event_id": "...",
      "error": "invalid timestamp"
    }
  ]
}
\end{lstlisting}
      \end{block}
    \end{column}
  \end{columns}
\end{frame}

\section{Storage}
% ══════════════════════════════════════════

\begin{frame}{Event Schema and Storage}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Unified Events Table}
      \begin{itemize}
        \item Single table with 10 columns
        \item Single Pydantic model (BehaviourEvent)
        \item Open metadata dict for extras
        \item 10 event types via ActivityKind enum
      \end{itemize}

      \vspace{0.2cm}
      \textbf{Event Types}\\
      {\small ENTRY, EXIT, REENTRY, ZONE\_ENTER,
      ZONE\_EXIT, ZONE\_DWELL, BILLING\_QUEUE\_JOIN,
      BILLING\_QUEUE\_LEAVE, BILLING\_QUEUE\_ABANDON,
      POS\_TRANSACTION}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{PostgreSQL 16}
      \begin{itemize}
        \item asyncpg + SQLAlchemy async
        \item PK idempotency on event\_id
        \item Indexes on (store\_id, timestamp) and (event\_type, store\_id, timestamp)
        \item O(log n) analytics queries
      \end{itemize}

      \vspace{0.2cm}
      \textbf{Test Isolation}
      \begin{itemize}
        \item SQLite (aiosqlite) per test
        \item Same schema on both dialects
        \item Zero-setup, in-memory
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\section{Design Decisions}
% ══════════════════════════════════════════

\begin{frame}{Key Design Decisions}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{YOLOv8-m over YOLOv8-L}
        CPU-friendly. All 5 clips finish in
        $\sim$46 min without a GPU.
      \end{block}

      \vspace{0.15cm}
      \begin{block}{Unified Events Table}
        Every query joins across event types.
        Split tables would need UNION ALL everywhere.
      \end{block}

      \vspace{0.15cm}
      \begin{block}{HSV Staff Detection}
        Near-zero cost vs CLIP zero-shot.
        Two signals: colour + dwell pattern.
      \end{block}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{WebSocket over SSE}
        Serves both web and terminal dashboards.
        Native Python client support.
      \end{block}

      \vspace{0.15cm}
      \begin{block}{Bounded Re-entry Cache}
        5-min window. O(live candidates) cost,
        not O(all history).
      \end{block}

      \vspace{0.15cm}
      \begin{block}{Two-Phase Abandon}
        Pipeline emits best-guess. API reclassifies
        after POS ingest for accuracy.
      \end{block}
    \end{column}
  \end{columns}
\end{frame}
